        let isAnimating = false;

        function openJar() {
            // Eğer halihazırda bir animasyon oynuyorsa tekrar tıklamayı engelle
            if (isAnimating) return;
            isAnimating = true;

            const jar = document.getElementById('jar');
            const paper = document.getElementById('paper');
            const messageText = document.getElementById('message-text');

            // Eğer kağıt zaten açıksa, önce onu gizle
            paper.classList.remove('pop-out');
            
            // Kavanoza sallanma efekti ver
            jar.classList.add('shake');

            // Sallanma bittikten sonra
            setTimeout(() => {
                jar.classList.remove('shake');
                
                // Rastgele mesaj seç
                const randomIndex = Math.floor(Math.random() * messages.length);
                messageText.innerHTML = messages[randomIndex];

                // Kağıdı fırlat
                paper.classList.add('pop-out');
                
                // Animasyon kilidini aç
                setTimeout(() => {
                    isAnimating = false;
                }, 1000);

            }, 500); // 0.5 saniye sallanma süresi
        }
